3use crate::concept::Concept;
4use crate::language::Language;
5use crate::registry::Registry;

8/// A search query against the registry.
9#[derive(Debug, Clone, Serialize, Deserialize)]
10pub struct SearchQuery {
11    /// Free-text search (matched against title, description, tags).
12    pub text: Option<String>,
13    /// Filter by language (only concepts with this language's example).
14    pub language: Option<Language>,
15    /// Filter by tags (all must match).
16    pub tags: Vec<String>,
17    /// Maximum results to return.
18    pub limit: Option<usize>,
19}
20
21impl SearchQuery {
22    /// Create a simple text search.
23    #[must_use]
24    pub fn text(query: impl Into<String>) -> Self {
25        Self {
26            text: Some(query.into()),
27            language: None,
28            tags: vec![],
29            limit: None,
30        }
31    }
32
33    /// Create a tag-based search.
34    #[must_use]
35    pub fn tagged(tags: Vec<String>) -> Self {
36        Self {
37            text: None,
38            language: None,
39            tags,
40            limit: None,
41        }
42    }
43
44    /// Filter results to concepts that have an example in this language.
45    #[must_use]
46    pub fn with_language(mut self, lang: Language) -> Self {
47        self.language = Some(lang);
48        self
49    }
50
51    /// Limit the number of results returned.
52    #[must_use]
53    pub fn with_limit(mut self, limit: usize) -> Self {
54        self.limit = Some(limit);
55        self
56    }
57
58    /// Add required tags (all must match).
59    #[must_use]
60    pub fn with_tags(mut self, tags: Vec<String>) -> Self {
61        self.tags = tags;
62        self
63    }
64}

66/// A search result with relevance score.
67#[derive(Debug, Clone, Serialize, Deserialize)]
68pub struct SearchResult {
69    /// Concept ID.
70    pub id: String,
71    /// Concept title.
72    pub title: String,
73    /// Relevance score (higher = better match).
74    pub score: f32,
75}
76
77/// Execute a search against the registry.
78#[must_use]
79pub fn search(registry: &Registry, query: &SearchQuery) -> Vec<SearchResult> {
80    let mut results: Vec<SearchResult> = registry
81        .list()
82        .into_iter()
83        .filter_map(|concept| score_concept(concept, query))
84        .collect();
85
86    results.sort_by(|a, b| {
87        b.score
88            .partial_cmp(&a.score)
89            .unwrap_or(std::cmp::Ordering::Equal)
90    });
91
92    if let Some(limit) = query.limit {
93        results.truncate(limit);
94    }
95
96    results
97}

99fn score_concept(concept: &Concept, query: &SearchQuery) -> Option<SearchResult> {
100    let mut score: f32 = 0.0;
101
102    // Language filter (hard filter, not scoring)
103    if let Some(lang) = query.language {
104        concept.example(lang)?;
105    }
106
107    // Tag filter (all must match)
108    for tag in &query.tags {
109        if !concept.has_tag(tag) {
110            return None;
111        }
112        score += 1.0;
113    }
114
115    // Text matching
116    if let Some(text) = &query.text {
117        let lower = text.to_lowercase();
118        let tokens: Vec<&str> = lower.split_whitespace().collect();
119        let score_before_text = score;
120
121        let id_lower = concept.id.to_lowercase();
122        let title_lower = concept.title.to_lowercase();
123        let desc_lower = concept.description.to_lowercase();
124
125        for token in &tokens {
126            // Exact ID match is the strongest signal
127            if id_lower == *token {
128                score += 5.0;
129            } else if id_lower.contains(token) {
130                score += 3.0;
131            }
132
133            // Title: exact match bonus, then substring
134            if title_lower == *token {
135                score += 4.0;
136            } else if title_lower.contains(token) {
137                score += 2.0;
138            }
139
140            // Description match
141            if desc_lower.contains(token) {
142                score += 1.0;
143            }
144
145            // Tag match — exact tag match is stronger than substring
146            for tag in &concept.tags {
147                let tag_lower = tag.to_lowercase();
148                if tag_lower == *token {
149                    score += 2.0;
150                } else if tag_lower.contains(token) {
151                    score += 1.0;
152                }
153            }
154        }
155
156        // Text was provided but didn't match anything — filter out
157        if score == score_before_text {
158            return None;
159        }
160    }
161
162    // If no query criteria at all, return everything with base score
163    if query.text.is_none() && query.tags.is_empty() && query.language.is_none() {
164        score = 1.0;
165    }
166
167    Some(SearchResult {
168        id: concept.id.clone(),
169        title: concept.title.clone(),
170        score,
171    })
172}

180    fn make_registry() -> Registry {
181        let mut reg = Registry::new();
182
183        let mut ex = HashMap::new();
184        ex.insert(
185            Language::Rust,
186            Example {
187                language: Language::Rust,
188                code: "let s = String::new();".into(),
189                explanation: "Rust strings".into(),
190                source_path: None,
191            },
192        );
193        reg.register(Concept {
194            id: "strings".into(),
195            title: "String Handling".into(),
196            topic: Topic::DataTypes,
197            description: "Working with text and string types.".into(),
198            best_practices: vec![],
199            gotchas: vec![],
200            performance_notes: vec![],
201            tags: vec!["text".into(), "utf-8".into()],
202            examples: ex,
203        });
204
205        reg.register(Concept {
206            id: "concurrency".into(),
207            title: "Concurrency Patterns".into(),
208            topic: Topic::Concurrency,
209            description: "Threads, async, and parallelism.".into(),
210            best_practices: vec![],
211            gotchas: vec![],
212            performance_notes: vec![],
213            tags: vec!["threads".into(), "async".into()],
214            examples: HashMap::new(),
