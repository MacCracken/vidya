1//! Supported programming languages and their metadata.
2//!
3//! Each language tracks its file extension, comment syntax, and how to
4//! compile/run examples for validation.
5
6use serde::{Deserialize, Serialize};
7
8/// A programming language supported by vidya.
9#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
10#[non_exhaustive]
11pub enum Language {
12    /// Rust (`.rs`) — compiled, systems programming
13    Rust,
14    /// Python (`.py`) — interpreted, general purpose
15    Python,
16    /// C (`.c`) — compiled, systems programming
17    C,
18    /// Go (`.go`) — compiled, concurrent systems
19    Go,
20    /// TypeScript (`.ts`) — transpiled, web/server
21    TypeScript,
22    /// Shell (`.sh`) — interpreted, scripting/automation
23    Shell,
24    /// Zig (`.zig`) — compiled, systems programming
25    Zig,
26}
27
28impl Language {
29    /// File extension for this language (without dot).
30    #[must_use]
31    pub const fn extension(&self) -> &'static str {
32        match self {
33            Self::Rust => "rs",
34            Self::Python => "py",
35            Self::C => "c",
36            Self::Go => "go",
37            Self::TypeScript => "ts",
38            Self::Shell => "sh",
39            Self::Zig => "zig",
40        }
41    }
42
43    /// Human-readable display name.
44    #[must_use]
45    pub const fn display_name(&self) -> &'static str {
46        match self {
47            Self::Rust => "Rust",
48            Self::Python => "Python",
49            Self::C => "C",
50            Self::Go => "Go",
51            Self::TypeScript => "TypeScript",
52            Self::Shell => "Shell",
53            Self::Zig => "Zig",
54        }
55    }
56
57    /// Single-line comment prefix.
58    #[must_use]
59    pub const fn comment_prefix(&self) -> &'static str {
60        match self {
61            Self::Rust | Self::C | Self::Go | Self::TypeScript | Self::Zig => "//",
62            Self::Python | Self::Shell => "#",
63        }
64    }
65
66    /// All supported languages.
67    #[must_use]
68    pub const fn all() -> &'static [Language] {
69        &[
70            Self::Rust,
71            Self::Python,
72            Self::C,
73            Self::Go,
74            Self::TypeScript,
75            Self::Shell,
76            Self::Zig,
77        ]
78    }
79
80    /// Parse from a string (case-insensitive, accepts extensions too).
81    #[must_use]
82    pub fn from_str_loose(s: &str) -> Option<Self> {
83        match s.to_lowercase().as_str() {
84            "rust" | "rs" => Some(Self::Rust),
85            "python" | "py" => Some(Self::Python),
86            "c" => Some(Self::C),
87            "go" | "golang" => Some(Self::Go),
88            "typescript" | "ts" => Some(Self::TypeScript),
89            "shell" | "sh" | "bash" | "zsh" => Some(Self::Shell),
90            "zig" => Some(Self::Zig),
91            _ => None,
92        }
93    }
94}
95
96impl std::fmt::Display for Language {
97    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
98        f.write_str(self.display_name())
99    }
100}

102#[cfg(test)]
103mod tests {
104    use super::*;
105
106    #[test]
107    fn all_languages() {
108        let all = Language::all();
109        assert_eq!(all.len(), 7);
110    }
111
112    #[test]
113    fn extensions() {
114        assert_eq!(Language::Rust.extension(), "rs");
115        assert_eq!(Language::Python.extension(), "py");
116        assert_eq!(Language::C.extension(), "c");
117        assert_eq!(Language::Go.extension(), "go");
118        assert_eq!(Language::TypeScript.extension(), "ts");
119        assert_eq!(Language::Shell.extension(), "sh");
120        assert_eq!(Language::Zig.extension(), "zig");
121    }
122
123    #[test]
124    fn from_str_loose_variants() {
125        assert_eq!(Language::from_str_loose("rust"), Some(Language::Rust));
126        assert_eq!(Language::from_str_loose("RS"), Some(Language::Rust));
127        assert_eq!(Language::from_str_loose("golang"), Some(Language::Go));
128        assert_eq!(Language::from_str_loose("bash"), Some(Language::Shell));
129        assert_eq!(Language::from_str_loose("haskell"), None);
130    }
131
132    #[test]
133    fn display() {
134        assert_eq!(Language::Rust.to_string(), "Rust");
135        assert_eq!(Language::TypeScript.to_string(), "TypeScript");
136    }
137
138    #[test]
139    fn comment_prefix() {
140        assert_eq!(Language::Rust.comment_prefix(), "//");
141        assert_eq!(Language::Python.comment_prefix(), "#");
142    }
143
144    #[test]
145    fn serde_roundtrip() {
146        let lang = Language::Rust;
147        let json = serde_json::to_string(&lang).unwrap();
148        let decoded: Language = serde_json::from_str(&json).unwrap();
149        assert_eq!(lang, decoded);
150    }
