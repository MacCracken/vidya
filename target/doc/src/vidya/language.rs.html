1//! Supported programming languages and their metadata.
2//!
3//! Each language tracks its file extension, comment syntax, and how to
4//! compile/run examples for validation.
5
6use serde::{Deserialize, Serialize};
7
8/// A programming language supported by vidya.
9#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
10#[non_exhaustive]
11pub enum Language {
12    /// Rust (`.rs`) — compiled, systems programming
13    Rust,
14    /// Python (`.py`) — interpreted, general purpose
15    Python,
16    /// C (`.c`) — compiled, systems programming
17    C,
18    /// Go (`.go`) — compiled, concurrent systems
19    Go,
20    /// TypeScript (`.ts`) — transpiled, web/server
21    TypeScript,
22    /// Shell (`.sh`) — interpreted, scripting/automation
23    Shell,
24    /// Zig (`.zig`) — compiled, systems programming
25    Zig,
26    /// x86_64 Assembly (`.s`) — GNU as syntax, 64-bit x86
27    AsmX86_64,
28    /// AArch64 Assembly (`.s`) — GNU as syntax, 64-bit ARM
29    AsmAarch64,
30    /// OpenQASM (`.qasm`) — quantum circuit assembly language
31    OpenQASM,
32}
33
34impl Language {
35    /// File extension for this language (without dot).
36    #[must_use]
37    pub const fn extension(&self) -> &'static str {
38        match self {
39            Self::Rust => "rs",
40            Self::Python => "py",
41            Self::C => "c",
42            Self::Go => "go",
43            Self::TypeScript => "ts",
44            Self::Shell => "sh",
45            Self::Zig => "zig",
46            Self::AsmX86_64 => "s",
47            Self::AsmAarch64 => "s",
48            Self::OpenQASM => "qasm",
49        }
50    }
51
52    /// Human-readable display name.
53    #[must_use]
54    pub const fn display_name(&self) -> &'static str {
55        match self {
56            Self::Rust => "Rust",
57            Self::Python => "Python",
58            Self::C => "C",
59            Self::Go => "Go",
60            Self::TypeScript => "TypeScript",
61            Self::Shell => "Shell",
62            Self::Zig => "Zig",
63            Self::AsmX86_64 => "x86_64 Assembly",
64            Self::AsmAarch64 => "AArch64 Assembly",
65            Self::OpenQASM => "OpenQASM",
66        }
67    }
68
69    /// Filename stem for content files (e.g. "rust", "python", "asm_x86_64").
70    ///
71    /// Combined with [`Self::extension()`] to form the full filename.
72    #[must_use]
73    pub const fn file_stem(&self) -> &'static str {
74        match self {
75            Self::Rust => "rust",
76            Self::Python => "python",
77            Self::C => "c",
78            Self::Go => "go",
79            Self::TypeScript => "typescript",
80            Self::Shell => "shell",
81            Self::Zig => "zig",
82            Self::AsmX86_64 => "asm_x86_64",
83            Self::AsmAarch64 => "asm_aarch64",
84            Self::OpenQASM => "openqasm",
85        }
86    }
87
88    /// Single-line comment prefix.
89    #[must_use]
90    pub const fn comment_prefix(&self) -> &'static str {
91        match self {
92            Self::Rust | Self::C | Self::Go | Self::TypeScript | Self::Zig => "//",
93            Self::Python | Self::Shell => "#",
94            Self::AsmX86_64 | Self::AsmAarch64 => "#",
95            Self::OpenQASM => "//",
96        }
97    }
98
99    /// All supported languages.
100    #[must_use]
101    pub const fn all() -> &'static [Language] {
102        &[
103            Self::Rust,
104            Self::Python,
105            Self::C,
106            Self::Go,
107            Self::TypeScript,
108            Self::Shell,
109            Self::Zig,
110            Self::AsmX86_64,
111            Self::AsmAarch64,
112            Self::OpenQASM,
113        ]
114    }
115
116    /// Parse from a string (case-insensitive, accepts extensions too).
117    #[must_use]
118    pub fn from_str_loose(s: &str) -> Option<Self> {
119        match s.to_lowercase().as_str() {
120            "rust" | "rs" => Some(Self::Rust),
121            "python" | "py" => Some(Self::Python),
122            "c" => Some(Self::C),
123            "go" | "golang" => Some(Self::Go),
124            "typescript" | "ts" => Some(Self::TypeScript),
125            "shell" | "sh" | "bash" | "zsh" => Some(Self::Shell),
126            "zig" => Some(Self::Zig),
127            "asm_x86_64" | "x86_64" | "x86-64" | "amd64" => Some(Self::AsmX86_64),
128            "asm_aarch64" | "aarch64" | "arm64" => Some(Self::AsmAarch64),
129            "openqasm" | "qasm" | "quantum" => Some(Self::OpenQASM),
130            _ => None,
131        }
132    }
133}
134
135impl std::fmt::Display for Language {
136    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
137        f.write_str(self.display_name())
138    }
139}

141#[cfg(test)]
142mod tests {
143    use super::*;
144
145    #[test]
146    fn all_languages() {
147        let all = Language::all();
148        assert_eq!(all.len(), 10);
149    }
150
151    #[test]
152    fn extensions() {
153        assert_eq!(Language::Rust.extension(), "rs");
154        assert_eq!(Language::Python.extension(), "py");
155        assert_eq!(Language::C.extension(), "c");
156        assert_eq!(Language::Go.extension(), "go");
157        assert_eq!(Language::TypeScript.extension(), "ts");
158        assert_eq!(Language::Shell.extension(), "sh");
159        assert_eq!(Language::Zig.extension(), "zig");
160    }
161
162    #[test]
163    fn from_str_loose_variants() {
164        assert_eq!(Language::from_str_loose("rust"), Some(Language::Rust));
165        assert_eq!(Language::from_str_loose("RS"), Some(Language::Rust));
166        assert_eq!(Language::from_str_loose("golang"), Some(Language::Go));
167        assert_eq!(Language::from_str_loose("bash"), Some(Language::Shell));
168        assert_eq!(Language::from_str_loose("haskell"), None);
169    }
170
171    #[test]
172    fn display() {
173        assert_eq!(Language::Rust.to_string(), "Rust");
174        assert_eq!(Language::TypeScript.to_string(), "TypeScript");
175    }
176
177    #[test]
178    fn comment_prefix() {
179        assert_eq!(Language::Rust.comment_prefix(), "//");
180        assert_eq!(Language::Python.comment_prefix(), "#");
181    }
182
183    #[test]
184    fn serde_roundtrip() {
185        let lang = Language::Rust;
186        let json = serde_json::to_string(&lang).unwrap();
187        let decoded: Language = serde_json::from_str(&json).unwrap();
188        assert_eq!(lang, decoded);
189    }
190}
