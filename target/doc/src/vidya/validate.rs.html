1//! Example validation — compile/run verification.
2//!
3//! Validates that code examples actually work by invoking the appropriate
4//! compiler or interpreter. Used in CI to ensure every example in the
5//! content directory is correct.
6
7use crate::language::Language;
8use crate::registry::Registry;
9use serde::{Deserialize, Serialize};
10use std::path::Path;
11use std::time::Instant;
12
13/// Result of validating a single example.
14#[derive(Debug, Clone, Serialize, Deserialize)]
15pub struct ValidationResult {
16    /// Concept ID.
17    pub concept_id: String,
18    /// Language of the example.
19    pub language: Language,
20    /// Whether validation passed.
21    pub passed: bool,
22    /// Error message if validation failed.
23    pub error: Option<String>,
24    /// Duration of validation in milliseconds.
25    pub duration_ms: u64,
26}
27
28/// Validation command for a language.
29///
30/// Returns the shell command template to compile/run a source file.
31/// `{file}` is replaced with the actual file path.
32#[must_use]
33pub fn validation_command(lang: Language) -> Option<&'static str> {
34    match lang {
35        Language::Rust => Some("rustc --edition 2024 {file} -o /tmp/vidya_test && /tmp/vidya_test"),
36        Language::Python => Some("python3 -c \"exec(open('{file}').read())\""),
37        Language::C => {
38            Some("gcc -std=c11 -Wall -Werror {file} -o /tmp/vidya_test && /tmp/vidya_test")
39        }
40        Language::Go => Some("go run {file}"),
41        Language::TypeScript => Some("bun run {file}"),
42        Language::Shell => Some("bash -n {file}"),
43        Language::Zig => Some("zig build-exe {file} -o /tmp/vidya_test && /tmp/vidya_test"),
44    }
45}
46
47/// Run validation for a single source file.
48///
49/// Executes the language-appropriate compile/run command and captures
50/// the result. Returns a [`ValidationResult`] indicating pass/fail.
51pub fn run_validation(concept_id: &str, language: Language, file_path: &Path) -> ValidationResult {
52    let start = Instant::now();
53
54    let Some(cmd_template) = validation_command(language) else {
55        return ValidationResult {
56            concept_id: concept_id.to_string(),
57            language,
58            passed: false,
59            error: Some("no validation command for this language".into()),
60            duration_ms: 0,
61        };
62    };
63
64    let file_str = file_path.display().to_string();
65    let cmd = cmd_template.replace("{file}", &file_str);
66
67    let result = std::process::Command::new("sh")
68        .args(["-c", &cmd])
69        .stdout(std::process::Stdio::piped())
70        .stderr(std::process::Stdio::piped())
71        .output();
72
73    let duration_ms = start.elapsed().as_millis() as u64;
74
75    match result {
76        Ok(output) => {
77            if output.status.success() {
78                ValidationResult {
79                    concept_id: concept_id.to_string(),
80                    language,
81                    passed: true,
82                    error: None,
83                    duration_ms,
84                }
85            } else {
86                let stderr = String::from_utf8_lossy(&output.stderr);
87                let stdout = String::from_utf8_lossy(&output.stdout);
88                let msg = if stderr.is_empty() {
89                    stdout.into_owned()
90                } else {
91                    stderr.into_owned()
92                };
93                ValidationResult {
94                    concept_id: concept_id.to_string(),
95                    language,
96                    passed: false,
97                    error: Some(msg),
98                    duration_ms,
99                }
100            }
101        }
102        Err(e) => ValidationResult {
103            concept_id: concept_id.to_string(),
104            language,
105            passed: false,
106            error: Some(format!("failed to execute command: {e}")),
107            duration_ms,
108        },
109    }
110}

112/// Validate all examples in a registry against their source files.
113///
114/// `content_dir` is the root content directory (e.g. `content/`).
115/// Each concept's examples are validated using their `source_path` field.
116pub fn validate_all(registry: &Registry, content_dir: &Path) -> Vec<ValidationResult> {
117    let mut results = Vec::new();
118
119    for concept in registry.list() {
120        for (lang, example) in &concept.examples {
121            let Some(ref source_path) = example.source_path else {
122                continue;
123            };
124            let file_path = content_dir.join(source_path);
125            if !file_path.exists() {
126                results.push(ValidationResult {
127                    concept_id: concept.id.clone(),
128                    language: *lang,
129                    passed: false,
130                    error: Some(format!("source file not found: {}", file_path.display())),
131                    duration_ms: 0,
132                });
133                continue;
134            }
135            results.push(run_validation(&concept.id, *lang, &file_path));
136        }
137    }
138
139    results
140}

142#[cfg(test)]
143mod tests {
144    use super::*;
145
146    #[test]
147    fn validation_commands_exist() {
148        for lang in Language::all() {
149            assert!(
150                validation_command(*lang).is_some(),
151                "missing validation command for {lang}"
152            );
153        }
154    }
155
156    #[test]
157    fn validation_result_serde() {
158        let result = ValidationResult {
159            concept_id: "strings".into(),
160            language: Language::Rust,
161            passed: true,
162            error: None,
163            duration_ms: 42,
164        };
165        let json = serde_json::to_string(&result).unwrap();
166        let decoded: ValidationResult = serde_json::from_str(&json).unwrap();
167        assert_eq!(decoded.concept_id, "strings");
168        assert!(decoded.passed);
169    }
170
171    #[test]
172    fn validation_result_failure() {
173        let result = ValidationResult {
174            concept_id: "concurrency".into(),
175            language: Language::C,
176            passed: false,
177            error: Some("segfault".into()),
178            duration_ms: 100,
179        };
180        assert!(!result.passed);
181        assert_eq!(result.error.as_deref(), Some("segfault"));
182    }
